```js static
import Fieldset from '~components/Fieldset'

function DisplayFieldset() {
  return (
    <Fieldset legend="legend" center>
      Some Text
    </Fieldset>
  )
}
```

```js
<Fieldset>Fieldset elements</Fieldset>

```

```js
<Fieldset center legend="legend">
  Fieldset elements
</Fieldset>
```

Fieldset aligned left:

